# Go SDK Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Go SDK for Roche that provides Client + Sandbox abstractions over dual-transport (gRPC/CLI) with Go-idiomatic patterns.

**Architecture:** Transport interface with two implementations (gRPC, CLI). Client auto-detects daemon via `~/.roche/daemon.json`, falls back to CLI. Sandbox wraps transport with instance context. Sentinel errors for typed error handling.

**Tech Stack:** Go 1.21+, google.golang.org/grpc, google.golang.org/protobuf, protoc-gen-go + protoc-gen-go-grpc

**Spec:** `docs/superpowers/specs/2026-03-15-go-sdk-design.md`

```go
package roche

// SandboxConfig configures a new sandbox.
// Zero values get safe defaults applied at the transport layer:
// Provider="docker", Image="python:3.12-slim", TimeoutSecs=300.
type SandboxConfig struct {
	Provider    string
	Image       string
	Memory      string  // e.g. "512m", "1g"
	CPUs        float64 // e.g. 1.5
	TimeoutSecs uint64  // 0 = use server default (300); cannot express "no timeout"
	Network     bool    // default false (AI-safe)
	Writable    bool    // default false (AI-safe)
	Env         map[string]string
	Mounts      []Mount
	Kernel      string // Firecracker only
	Rootfs      string // Firecracker only
}

// Mount configures a host directory mount.
// Use NewMount() to create mounts with AI-safe defaults (Readonly=true).
type Mount struct {
	HostPath      string
	ContainerPath string
	Readonly      bool
}

// NewMount creates a Mount with AI-safe defaults (Readonly=true).
func NewMount(hostPath, containerPath string) Mount {
	return Mount{HostPath: hostPath, ContainerPath: containerPath, Readonly: true}
}

// ExecOutput contains the result of executing a command.
type ExecOutput struct {
	ExitCode int32
	Stdout   string
	Stderr   string
}

// SandboxStatus represents the state of a sandbox.
type SandboxStatus string

const (
	StatusRunning SandboxStatus = "running"
	StatusPaused  SandboxStatus = "paused"
	StatusStopped SandboxStatus = "stopped"
	StatusFailed  SandboxStatus = "failed"
)

// SandboxInfo contains metadata about a sandbox.
type SandboxInfo struct {
	ID        string
	Status    SandboxStatus
	Provider  string
	Image     string
	ExpiresAt *uint64 // nil if no expiry
}

// GCOptions configures garbage collection behavior.
type GCOptions struct {
	DryRun bool
	All    bool
}

// PoolInfo contains metadata about a sandbox pool.
type PoolInfo struct {
	Provider    string
	Image       string
	IdleCount   uint32
	ActiveCount uint32
	MaxIdle     uint32
	MaxTotal    uint32
}

// PoolConfig configures a sandbox pool for warmup.
type PoolConfig struct {
	Provider string
	Image    string
	Count    uint32
}
```

```go
const (
	defaultImage       = "python:3.12-slim"
	defaultTimeoutSecs = 300
	defaultProvider    = "docker"
	defaultBinary      = "roche"
)

// applyDefaults fills zero-value fields in cfg with safe defaults.
// clientProvider is the client-level default provider.
func applyDefaults(cfg SandboxConfig, clientProvider string) SandboxConfig {
	if cfg.Provider == "" {
		cfg.Provider = clientProvider
	}
	if cfg.Image == "" {
		cfg.Image = defaultImage
	}
	if cfg.TimeoutSecs == 0 {
		cfg.TimeoutSecs = defaultTimeoutSecs
	}
	return cfg
}
```

```go
package roche

import "context"

// Transport defines the low-level communication protocol with the Roche backend.
type Transport interface {
	Create(ctx context.Context, cfg SandboxConfig, provider string) (string, error)
	Exec(ctx context.Context, sandboxID string, command []string, provider string, timeoutSecs *uint64) (*ExecOutput, error)
	Destroy(ctx context.Context, sandboxIDs []string, provider string, all bool) ([]string, error)
	List(ctx context.Context, provider string) ([]SandboxInfo, error)
	Pause(ctx context.Context, sandboxID, provider string) error
	Unpause(ctx context.Context, sandboxID, provider string) error
	GC(ctx context.Context, provider string, dryRun, all bool) ([]string, error)
	CopyTo(ctx context.Context, sandboxID, hostPath, sandboxPath, provider string) error
	CopyFrom(ctx context.Context, sandboxID, sandboxPath, hostPath, provider string) error
	PoolStatus(ctx context.Context) ([]PoolInfo, error)
	PoolWarmup(ctx context.Context, pools []PoolConfig) error
	PoolDrain(ctx context.Context, provider, image string) error
	Close() error
}
```

```go
package roche

import (
	"context"
	"encoding/json"
	"errors"
	"fmt"
	"os/exec"
	"strconv"
	"strings"
)

type cliTransport struct {
	binary string
}

func newCLITransport(binary string) *cliTransport {
	return &cliTransport{binary: binary}
}

func (c *cliTransport) Create(ctx context.Context, cfg SandboxConfig, provider string) (string, error) {
	args := c.buildCreateArgs(cfg, provider)
	stdout, stderr, err := c.run(ctx, args)
	if err != nil {
		if cliErr := mapCLIError(stderr); cliErr != nil {
			return "", cliErr
		}
		return "", fmt.Errorf("roche create failed: %s", stderr)
	}
	return strings.TrimSpace(stdout), nil
}

func (c *cliTransport) Exec(ctx context.Context, sandboxID string, command []string, provider string, timeoutSecs *uint64) (*ExecOutput, error) {
	args := c.buildExecArgs(sandboxID, command, provider, timeoutSecs)
	stdout, stderr, exitCode, err := c.runUnchecked(ctx, args)
	if err != nil {
		if errors.Is(err, exec.ErrNotFound) {
			return nil, fmt.Errorf("roche binary not found: %w", ErrUnavailable)
		}
		// Check if it's a Roche-level error (not a command exit code)
		if cliErr := mapCLIError(stderr); cliErr != nil {
			return nil, cliErr
		}
	}
	return &ExecOutput{
		ExitCode: int32(exitCode),
		Stdout:   stdout,
		Stderr:   stderr,
	}, nil
}

func (c *cliTransport) Destroy(ctx context.Context, sandboxIDs []string, provider string, all bool) ([]string, error) {
	args := c.buildDestroyArgs(sandboxIDs, provider, all)
	stdout, stderr, err := c.run(ctx, args)
	if err != nil {
		if cliErr := mapCLIError(stderr); cliErr != nil {
			return nil, cliErr
		}
		return nil, fmt.Errorf("roche destroy failed: %s", stderr)
	}
	return splitNonEmpty(stdout), nil
}

func (c *cliTransport) List(ctx context.Context, provider string) ([]SandboxInfo, error) {
	args := c.buildListArgs(provider)
	stdout, stderr, err := c.run(ctx, args)
	if err != nil {
		if cliErr := mapCLIError(stderr); cliErr != nil {
			return nil, cliErr

func (c *cliTransport) run(ctx context.Context, args []string) (stdout, stderr string, err error) {
	cmd := exec.CommandContext(ctx, c.binary, args...)
	var outBuf, errBuf strings.Builder
	cmd.Stdout = &outBuf
	cmd.Stderr = &errBuf
	err = cmd.Run()
	if errors.Is(err, exec.ErrNotFound) {
		return "", "", fmt.Errorf("roche binary %q not found: %w", c.binary, ErrUnavailable)
	}
	return outBuf.String(), errBuf.String(), err
}

func (c *cliTransport) runUnchecked(ctx context.Context, args []string) (stdout, stderr string, exitCode int, err error) {
	cmd := exec.CommandContext(ctx, c.binary, args...)
	var outBuf, errBuf strings.Builder
	cmd.Stdout = &outBuf
	cmd.Stderr = &errBuf
	err = cmd.Run()
	exitCode = cmd.ProcessState.ExitCode()
	return outBuf.String(), errBuf.String(), exitCode, err
}

func mapCLIError(stderr string) error {
	if !strings.HasPrefix(stderr, "Error: ") {
		return nil
	}
	lower := strings.ToLower(stderr)
	switch {
	case strings.Contains(lower, "not found"):
		return fmt.Errorf("%s: %w", strings.TrimSpace(stderr), ErrNotFound)
	case strings.Contains(lower, "paused"):
		return fmt.Errorf("%s: %w", strings.TrimSpace(stderr), ErrPaused)
	case strings.Contains(lower, "unavailable"), strings.Contains(lower, "connection refused"):
		return fmt.Errorf("%s: %w", strings.TrimSpace(stderr), ErrUnavailable)
	case strings.Contains(lower, "timeout"), strings.Contains(lower, "timed out"):
		return fmt.Errorf("%s: %w", strings.TrimSpace(stderr), ErrTimeout)
	case strings.Contains(lower, "unsupported"):
		return fmt.Errorf("%s: %w", strings.TrimSpace(stderr), ErrUnsupported)
	default:
		return fmt.Errorf("roche: %s", strings.TrimSpace(stderr))
	}
}

func splitNonEmpty(s string) []string {
	var result []string
	for _, line := range strings.Split(strings.TrimSpace(s), "\n") {
		if line != "" {
			result = append(result, line)
		}
	}
	return result
}
```

```go
package roche

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"syscall"
)

type daemonInfo struct {
	PID  int `json:"pid"`
	Port int `json:"port"`
}

// detectDaemon reads ~/.roche/daemon.json and verifies the process is alive.
func detectDaemon() (*daemonInfo, error) {
	home, err := os.UserHomeDir()
	if err != nil {
		return nil, fmt.Errorf("cannot determine home directory: %w", err)
	}
	path := filepath.Join(home, ".roche", "daemon.json")
	info, err := parseDaemonFile(path)
	if err != nil {
		return nil, err
	}
	if !isProcessAlive(info.PID) {
		return nil, fmt.Errorf("daemon process %d is not alive", info.PID)
	}
	return info, nil
}

func parseDaemonFile(path string) (*daemonInfo, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("cannot read daemon file: %w", err)
	}
	var info daemonInfo
	if err := json.Unmarshal(data, &info); err != nil {
		return nil, fmt.Errorf("cannot parse daemon file: %w", err)
	}
	if info.Port == 0 {
		return nil, fmt.Errorf("daemon file has invalid port: %d", info.Port)
	}
	if info.PID == 0 {
		return nil, fmt.Errorf("daemon file has invalid pid: %d", info.PID)
	}
	return &info, nil
}

func isProcessAlive(pid int) bool {
	process, err := os.FindProcess(pid)
	if err != nil {
		return false
	}
	err = process.Signal(syscall.Signal(0))
	return err == nil
}
```

```go
func (g *grpcTransport) getClient() (pb.SandboxServiceClient, error) {
	g.mu.Lock()
	defer g.mu.Unlock()
	if g.client != nil {
		return g.client, nil
	}
	addr := fmt.Sprintf("localhost:%d", g.port)
	conn, err := grpc.NewClient(addr, grpc.WithTransportCredentials(insecure.NewCredentials()))
	if err != nil {
		return nil, fmt.Errorf("failed to connect to daemon at %s: %w", addr, ErrUnavailable)
	}
	g.conn = conn
	g.client = pb.NewSandboxServiceClient(conn)
	return g.client, nil
}

	// Note: PoolDrainRequest fields depend on the proto definition.
	// The CLI `pool drain` triggers a global drain.
	_, err = client.PoolDrain(ctx, &pb.PoolDrainRequest{})
	if err != nil {
		return mapGRPCError(err)
	}
	return nil
}

func (g *grpcTransport) Close() error {
	g.mu.Lock()
	defer g.mu.Unlock()
	if g.conn != nil {
		return g.conn.Close()
	}
	return nil
}

// --- Helpers ---

func mapGRPCError(err error) error {
	if err == nil {
		return nil
	}
	st, ok := status.FromError(err)
	if !ok {
		return err
	}
	msg := st.Message()
	switch st.Code() {
	case codes.NotFound:
		return fmt.Errorf("%s: %w", msg, ErrNotFound)
	case codes.FailedPrecondition:
		return fmt.Errorf("%s: %w", msg, ErrPaused)
	case codes.Unavailable:
		return fmt.Errorf("%s: %w", msg, ErrUnavailable)
	case codes.DeadlineExceeded:
		return fmt.Errorf("%s: %w", msg, ErrTimeout)
	case codes.Unimplemented:
		return fmt.Errorf("%s: %w", msg, ErrUnsupported)
	default:
		return fmt.Errorf("roche: gRPC error: %s", msg)
	}
}

func mapProtoStatus(s pb.SandboxStatus) SandboxStatus {
	switch s {
	case pb.SandboxStatus_SANDBOX_STATUS_RUNNING:
		return StatusRunning
	case pb.SandboxStatus_SANDBOX_STATUS_PAUSED:
		return StatusPaused
	case pb.SandboxStatus_SANDBOX_STATUS_STOPPED:
		return StatusStopped
	case pb.SandboxStatus_SANDBOX_STATUS_FAILED:
		return StatusFailed
	default:
		return StatusFailed
	}
}
```

```go
package roche

import "context"

// Client is the main entry point for interacting with Roche sandboxes.
type Client struct {
	transport Transport
	provider  string
}

// Option configures the Client.
type Option func(*clientConfig)

type clientConfig struct {
	transport  Transport
	binary     string
	daemonPort int
	provider   string
	directMode bool
}

// WithTransport injects a custom transport.
func WithTransport(t Transport) Option {
	return func(c *clientConfig) { c.transport = t }
}

// WithBinary sets the CLI binary path (default: "roche").
func WithBinary(path string) Option {
	return func(c *clientConfig) { c.binary = path }
}

// WithDaemonPort overrides the daemon port for gRPC transport.
func WithDaemonPort(port int) Option {
	return func(c *clientConfig) { c.daemonPort = port }
}

// WithProvider sets the default provider (default: "docker").
func WithProvider(provider string) Option {
	return func(c *clientConfig) { c.provider = provider }
}

// WithDirectMode forces CLI transport, bypassing daemon auto-detection.
func WithDirectMode() Option {
	return func(c *clientConfig) { c.directMode = true }
}

// New creates a new Roche client.
func New(opts ...Option) (*Client, error) {
	cfg := &clientConfig{
		binary:   defaultBinary,
		provider: defaultProvider,
	}
	for _, opt := range opts {
		opt(cfg)
	}

	if cfg.transport != nil {
		return &Client{transport: cfg.transport, provider: cfg.provider}, nil
	}

	if cfg.directMode {
		return &Client{
			transport: newCLITransport(cfg.binary),
			provider:  cfg.provider,
		}, nil
	}

	// Auto-detect daemon
	if info, err := detectDaemon(); err == nil {
		port := info.Port
		if cfg.daemonPort > 0 {
			port = cfg.daemonPort
		}
		return &Client{
			transport: newGRPCTransport(port),
			provider:  cfg.provider,
		}, nil
	}

	// Fallback to CLI
	return &Client{
		transport: newCLITransport(cfg.binary),
		provider:  cfg.provider,
	}, nil
}

// Create creates a new sandbox and returns a Sandbox handle.
func (c *Client) Create(ctx context.Context, cfg SandboxConfig) (*Sandbox, error) {
	cfg = applyDefaults(cfg, c.provider)
	id, err := c.transport.Create(ctx, cfg, cfg.Provider)
	if err != nil {
		return nil, err
	}
	return &Sandbox{id: id, provider: cfg.Provider, transport: c.transport}, nil
}

// CreateID creates a new sandbox and returns only the sandbox ID.
func (c *Client) CreateID(ctx context.Context, cfg SandboxConfig) (string, error) {
	cfg = applyDefaults(cfg, c.provider)
	return c.transport.Create(ctx, cfg, cfg.Provider)
}

// Exec executes a command in the specified sandbox.
func (c *Client) Exec(ctx context.Context, sandboxID string, command []string) (*ExecOutput, error) {
	return c.transport.Exec(ctx, sandboxID, command, c.provider, nil)
}

// Destroy destroys the specified sandbox.
func (c *Client) Destroy(ctx context.Context, sandboxID string) error {
	_, err := c.transport.Destroy(ctx, []string{sandboxID}, c.provider, false)
	return err
}

// DestroyMany destroys multiple sandboxes at once.
func (c *Client) DestroyMany(ctx context.Context, sandboxIDs []string) ([]string, error) {
	return c.transport.Destroy(ctx, sandboxIDs, c.provider, false)
}

// List returns all active sandboxes.
func (c *Client) List(ctx context.Context) ([]SandboxInfo, error) {
	return c.transport.List(ctx, c.provider)
}

// GC garbage-collects expired sandboxes.
func (c *Client) GC(ctx context.Context, opts GCOptions) ([]string, error) {
	return c.transport.GC(ctx, c.provider, opts.DryRun, opts.All)
}

// PoolStatus returns the current status of sandbox pools.
func (c *Client) PoolStatus(ctx context.Context) ([]PoolInfo, error) {
	return c.transport.PoolStatus(ctx)
}

// PoolWarmup warms up sandbox pools.

	return c.transport.PoolWarmup(ctx, pools)
}

	return c.transport.PoolDrain(ctx, provider, image)
}

// Close shuts down the client and releases resources.
func (c *Client) Close() error {
	return c.transport.Close()
}
```

```go
package roche

import "context"

// Sandbox represents a running sandbox instance.
type Sandbox struct {
	id        string
	provider  string
	transport Transport
}

// ID returns the sandbox identifier.
func (s *Sandbox) ID() string { return s.id }

// Provider returns the provider name.
func (s *Sandbox) Provider() string { return s.provider }

// Exec executes a command in this sandbox.
func (s *Sandbox) Exec(ctx context.Context, command []string) (*ExecOutput, error) {
	return s.transport.Exec(ctx, s.id, command, s.provider, nil)
}

// Pause pauses this sandbox.
func (s *Sandbox) Pause(ctx context.Context) error {
	return s.transport.Pause(ctx, s.id, s.provider)
}

// Unpause resumes this sandbox.
func (s *Sandbox) Unpause(ctx context.Context) error {
	return s.transport.Unpause(ctx, s.id, s.provider)
}

// Destroy destroys this sandbox.
func (s *Sandbox) Destroy(ctx context.Context) error {
	_, err := s.transport.Destroy(ctx, []string{s.id}, s.provider, false)
	return err
}

// Close is an alias for Destroy, enabling defer-based cleanup.
func (s *Sandbox) Close(ctx context.Context) error {
	return s.Destroy(ctx)
}

// CopyTo copies a file from the host to this sandbox.
func (s *Sandbox) CopyTo(ctx context.Context, hostPath, sandboxPath string) error {
	return s.transport.CopyTo(ctx, s.id, hostPath, sandboxPath, s.provider)
}

// CopyFrom copies a file from this sandbox to the host.
func (s *Sandbox) CopyFrom(ctx context.Context, sandboxPath, hostPath string) error {
	return s.transport.CopyFrom(ctx, s.id, sandboxPath, hostPath, s.provider)
}
```
